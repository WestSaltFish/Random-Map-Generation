What is Random map

In video games, a random map is a map generated randomly by the computer, usually in strategy, roguelike games. Random maps are often the core of single and multiplayer gameplay, aside from story based campaigns that are often shipped with the game.

                                Map* MapGenerator::GenerateDungeonMap(uint row, uint col, uint rooms, uint tileWidth, uint tileHeight, int eSeed)
                                {
                                    Map* ret = new Map(row, col, tileWidth, tileHeight);
                                
                                    InitSeed(eSeed);
                                
                                    // Limit room nums
                                    rooms = rooms > row * col ? row * col : rooms;
                                
                                    // Get random position, and it should not be on the edge
                                    iPoint currentPos = { 0,0 };
                                
                                    currentPos.x = (rand() % (row - 2)) + 1;
                                
                                    currentPos.y = (rand() % (row - 2)) + 1;
                                
                                    CreateBaseMap(ret);
                                
                                    // Create dungeon
                                    DungeonMapBacktrack(ret, &rooms, currentPos);
                                
                                    return ret;
                                }

                                Map* MapGenerator::GenerateDungeonMapCA(uint row, uint col, uint tileWidth, uint tileHeight, int eSeed)
                                {
                                    Map* ret = new Map(row, col, tileWidth, tileHeight);
                                
                                    InitSeed(eSeed);
                                
                                    CreateBaseMapCA(ret);
                                
                                    // Optimize map with Cellular Automata algorithm
                                    LoopOptimizeMapCA(ret, 3);
                                
                                    // Find and get map areas
                                    vector> areas = FindAreasMapCA(ret);
                                
                                    // Connect areas
                                    ConnectAreasMapCA(ret, areas);
                                
                                    return ret;
                                }

                                void MapGenerator::InitSeed(int eSeed)
                                {
                                    uint seed;
                                
                                    // If exist eSeed, chage seed to decrypted eSeed
                                    if (eSeed != 0) seed = Decrypt(eSeed);
                                    // Else init seed with time
                                    else seed = time(NULL);
                                
                                    // Save new seed to file
                                    SaveSeed(seed);
                                
                                    // Init srand with seed
                                    srand(seed);
                                }
